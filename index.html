```javascript // Get the mood select element const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); // Function to fetch and display songs for the selected mood async function fetchSongs(mood) { try { const response = await fetch(`api/moods/${mood}`); const data = await response.json(); return data.songs.map(song => `

${song.title}

${song.artist}

`).join(''); } catch (error) { console.error('Error:', error); return ''; } } // Add event listener to show songs button showSongsButton.addEventListener('click', async () => { // Get the selected mood const selectedMood = moodSelect.value; // Fetch and display songs for the selected mood const songList = await fetchSongs(selectedMood); resultsDiv.innerHTML = songList; }); // Add event listener to mood select moodSelect.addEventListener('change', () => { // Clear the results div resultsDiv.innerHTML = ''; }); ```
