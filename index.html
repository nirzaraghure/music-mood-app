```javascript const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); const songTemplate = document.createElement('template'); songTemplate.innerHTML = `

{{title}}

{{artist}}

`; function renderSongs(data) { if (!data || !data.songs) return; const fragment = document.createDocumentFragment(); data.songs.forEach(song => { const clone = songTemplate.content.cloneNode(true); clone.querySelector('h2').textContent = song.title; clone.querySelector('p').textContent = song.artist; fragment.appendChild(clone); }); resultsDiv.innerHTML = ''; resultsDiv.appendChild(fragment); } async function fetchSongs(mood) { try { const response = await fetch(`api/moods/${mood}`); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } return response.json(); } catch (error) { console.error('Error:', error); return null; } } async function handleButtonClick() { const selectedMood = moodSelect.value; const data = await fetchSongs(selectedMood); renderSongs(data); } showSongsButton.addEventListener('click', handleButtonClick); moodSelect.addEventListener('change', handleButtonClick); // Optional: Debounce event listener for 'change' event to improve performance // moodSelect.addEventListener('change', () => { // setTimeout(handleButtonClick, 500); // adjust the delay as needed // }); ```
