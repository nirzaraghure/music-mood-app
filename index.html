```javascript const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); showSongsButton.addEventListener('click', async () => { const selectedMood = moodSelect.value; try { const response = await fetch(`api/moods/${selectedMood}`); const data = await response.json(); const songListTemplate = document.createElement('template'); songListTemplate.innerHTML = ` {{#songs}}

{{title}}

{{artist}}

{{/songs}} `; const songList = songListTemplate.content.cloneNode(true).children[0]; data.songs.forEach(song => { const songElement = songListTemplate.content.cloneNode(true).children[0]; songElement.querySelector('h2').textContent = song.title; songElement.querySelector('p').textContent = song.artist; songList.appendChild(songElement); }); resultsDiv.innerHTML = ''; resultsDiv.appendChild(songList); } catch (error) { console.error('Error:', error); } }); moodSelect.addEventListener('change', () => { resultsDiv.innerHTML = ''; }); ```
