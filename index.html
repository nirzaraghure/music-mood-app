```javascript const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); const songTemplate = document.createElement('template'); songTemplate.innerHTML = `

{{title}}

{{artist}}

`; // Function to render songs function renderSongs(data) { if (!data || !data.songs) return; const fragment = document.createDocumentFragment(); data.songs.forEach((song) => { const clone = songTemplate.content.cloneNode(true); clone.querySelector('h2').textContent = song.title; clone.querySelector('p').textContent = song.artist; fragment.appendChild(clone); }); resultsDiv.innerHTML = ''; resultsDiv.appendChild(fragment); } // Function to fetch songs async function fetchSongs(mood) { try { const response = await fetch(`api/moods/${mood}`); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } return response.json(); } catch (error) { console.error('Error:', error); return Promise.resolve(null); } } // Function to handle user interaction function handleUserInteraction(mood) { fetchSongs(mood).then(renderSongs); } // Event listeners showSongsButton.addEventListener('click', () => handleUserInteraction(moodSelect.value)); moodSelect.addEventListener('change', () => handleUserInteraction(moodSelect.value)); ```
