```javascript const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); const songTemplate = document.createElement('template'); songTemplate.innerHTML = `

{{title}}

{{artist}}

`; showSongsButton.addEventListener('click', async () => { const selectedMood = moodSelect.value; try { const response = await fetch(`api/moods/${selectedMood}`); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const data = await response.json(); resultsDiv.innerHTML = ''; data.songs.forEach(song => { const clone = songTemplate.content.cloneNode(true); clone.querySelector('h2').textContent = song.title; clone.querySelector('p').textContent = song.artist; resultsDiv.appendChild(clone); }); } catch (error) { console.error('Error:', error); } }); moodSelect.addEventListener('change', () => { resultsDiv.innerHTML = ''; showSongsButton.click(); // Simulate click event to fetch new data }); ```
