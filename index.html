```javascript // Get the mood select element const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); // Add event listener to show songs button showSongsButton.addEventListener('click', async () => { // Get the selected mood const selectedMood = moodSelect.value; try { // Send a GET request to the API with the selected mood const response = await fetch(`api/moods/${selectedMood}`); const data = await response.json(); // Display the song list in the results div const songList = data.songs.map(song => `

${song.title}

${song.artist}

`).join(''); resultsDiv.innerHTML = songList; } catch (error) { console.error('Error:', error); } }); // Add event listener to mood select moodSelect.addEventListener('change', () => { // Clear the results div resultsDiv.innerHTML = ''; }); ```
