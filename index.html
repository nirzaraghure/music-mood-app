```javascript const moodSelect = document.getElementById('mood'); const showSongsButton = document.getElementById('show-songs-button'); const resultsDiv = document.getElementById('results'); showSongsButton.addEventListener('click', async () => { try { const selectedMood = moodSelect.value; const response = await fetch(`api/moods/${selectedMood}`); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const data = await response.json(); const songList = data.songs.map(song => `

${song.title}

${song.artist}

`).join(''); resultsDiv.innerHTML = songList; } catch (error) { console.error('Error fetching songs:', error); resultsDiv.innerHTML = '

Error loading songs.

'; } }); moodSelect.addEventListener('change', () => { resultsDiv.innerHTML = ''; }); ```
